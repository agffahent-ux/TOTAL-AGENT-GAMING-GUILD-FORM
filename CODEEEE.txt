    <script>
        // Store references to elements
        const linkContainer = document.getElementById('linkContainer');
        const formContainer = document.getElementById('formContainer');
        const registrationForm = document.getElementById('registrationForm');
        const messageBox = document.getElementById('messageBox');
        const gatheredInfoSection = document.getElementById('gatheredInfoSection');
        const infoDisplay = document.getElementById('infoDisplay');

        // Function to reveal the form when the 'simple link' button is clicked
        function showForm() {
            linkContainer.classList.add('hidden');
            formContainer.classList.remove('hidden');
            // Scroll to the top of the form for better visibility
            formContainer.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }

        function handleSubmission(event) {
            // Prevent the default form submission behavior
            event.preventDefault(); 

            // 1. Collect form data
            const realName = document.getElementById('realName').value;
            const gameName = document.getElementById('gameName').value;
            const uid = document.getElementById('uid').value; 
            const phoneNumber = document.getElementById('phoneNumber').value.trim() || 'Not Provided';
            
            const mapTypeElement = document.querySelector('input[name="mapType"]:checked');
            const gamingExperienceElement = document.querySelector('input[name="gamingExperience"]:checked');
            const timingFlexibilityElement = document.querySelector('input[name="timingFlexibility"]:checked');
            
            // Should always have values due to 'required' but check for safety
            const mapType = mapTypeElement ? mapTypeElement.value : 'N/A';
            const gamingExperience = gamingExperienceElement ? gamingExperienceElement.value : 'N/A';
            const timingFlexibility = timingFlexibilityElement ? timingFlexibilityElement.value : 'N/A';

            // 2. Display brief success message
            messageBox.textContent = `Thank you, ${gameName}! Registration successful.`;
            messageBox.classList.remove('hidden');
            
            // Hide message after a short delay
            setTimeout(() => {
                messageBox.classList.add('hidden');
            }, 3000); // Hide after 3 seconds

            // 3. Hide the form
            formContainer.classList.add('hidden');

            // 4. Populate the gathered information section
            infoDisplay.innerHTML = `
                <p><span class="info-label">Full Name:</span><span class="info-value">${realName}</span></p>
                <p><span class="info-label">IGN:</span><span class="info-value">${gameName}</span></p>
                <p><span class="info-label">Game ID (UID):</span><span class="info-value">${uid}</span></p>
                <p><span class="info-label">Phone Number:</span><span class="info-value">${phoneNumber}</span></p>
                <p><span class="info-label">Experience:</span><span class="info-value">${gamingExperience}</span></p>
                <p><span class="info-label">Map Preference:</span><span class="info-value">${mapType}</span></p>
                <p><span class="info-label">Timing Flexibility:</span><span class="info-value">${timingFlexibility}</span></p>
            `;

            // 5. Show the gathered information section
            gatheredInfoSection.classList.remove('hidden');

             // Scroll to the gathered info section smoothly
            gatheredInfoSection.scrollIntoView({ behavior: 'smooth', block: 'center' });
        }

        function resetForm() {
            // Hide the gathered info section
            gatheredInfoSection.classList.add('hidden');
            
            // Clear the form fields
            registrationForm.reset(); 

            // Show the simple link container again
            linkContainer.classList.remove('hidden');

            // Scroll back to the link container
            linkContainer.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }
    </script>
